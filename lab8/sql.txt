DROP TABLE IF EXISTS PERSON;
CREATE TABLE PERSON
(
        CONTACT_ID SERIAL,
        NAME VARCHAR(50) NOT NULL,
        SIZE VARCHAR(50) NOT NULL,
        PLACE VARCHAR(50) NOT NULL,
        DATEOFBIRTH VARCHAR(50) NOT NULL,
	    USERN VARCHAR(50) NOT NULL,
        PRIMARY KEY (CONTACT_ID)
);
